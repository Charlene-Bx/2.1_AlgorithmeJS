// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

function calcDistance (a,b){
    let distance= Math.sqrt( ((b[0]-a[0])**2) + ((b[1]-a[1])**2) );
    return Math.round(distance*100)/100
}
